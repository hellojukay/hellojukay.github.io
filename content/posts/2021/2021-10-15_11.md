    my $s = shift;

2. perl 多线程消费
参考文档: https://metacpan.org/pod/Thread::Queue
```perl
use threads;
use Thread::Queue;

my $q = Thread::Queue->new();

sub work ($) {
    while(my $task = $q->dequeue()){
        # if $task is undef, it means the queue end
        last unless($task);
        
        # do somthing
    }
}

# enqueue
async {
    foreach(0..10) {
        $q->enqueue($_)
    }
    # Signal that there is no more work to be sent
    $q->end();
}

# create threads
foreach(0..4) {
    threads->create(\&work);
}

# wait all thread finish
foreach(threads->list()) {
    $_->join();
}
```

3. 统计 ip 访问，倒序结果
```perl
my %hash;
while(<>){
    if(/(\d+\.\d+\.\d+\.\d+)/){
        my $value = $hash{$1} ? $hash{$1} : 0;
        $hash{$1} = $value + 1;
    }
}

my @keys = sort { $hash{$a} <=> $hash{$b} } keys(%hash);
foreach my $key (reverse @keys) {
    printf("%-15s%s\n",$key,$hash{$key});
}
```
