第五步： jenkinsfile 配置,给以简单额配置如下
```yaml
pipeline {
    environment {
        git_cred = "licong-biz"
        image = "docker.hellojukay.cn/server"
    }
  agent {
    kubernetes {
      defaultContainer 'jnlp'
      yaml """
apiVersion: v1
kind: Pod
metadata:
  labels:
    some-label: some-label-value
spec:
  containers:
  - name: gradle
    image: docker.hellojukay.cn/java:8-jdk-alpine
    command:
    - sh
    tty: true
  - name: docker
    image: docker.hellojukay.cn/docker:18.09
    command: ['cat']
    tty: true
    volumeMounts:
    - name: dockersock
      mountPath: /var/run/docker.sock
  volumes:
  - name: dockersock
    hostPath:
      path: /var/run/docker.sock
"""
    }
  }
  stages {
    stage('gradle build') {
      steps {
        sh "mkdir -p sec-matrix-core"
        dir("sec-matrix-core") {
            git branch: "${env.BRANCH_NAME}",credentialsId: "${git_cred}", url: "${env.GIT_URL}"
            container('gradle') {
                sh 'cd backend && ./gradlew build'
            }
            container('docker') {
                sh """
                  ls -al 
                  cd  backend && docker build -t ${image}:${env.BUILD_ID} .
                """
            }
        }
      }
    }
  }
}
```
